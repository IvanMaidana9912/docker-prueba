# Instrucciones:

## ENVIROMENTS:
- MYSQLDB_HOST=
- MYSQLDB_ROOT_PASSWORD=
- MYSQLDB_DATEBASE=ivandb

- MYSQLDB_LOCAL_PORT=
- MYSQLDB_DOCKER_PORT=

- NODE_LOCAL_PORT=
- NODE_DOCKER_PORT=

## RUN:

- docker compose up

## STOP:

- docker compose down

## BUILD:

- docker compose up --build

## OTHERS COMMANDS:

- docker -t NAME-CONTAINER .
- docker -it NAME-CONTAINER .
- docker images -a
- docker rmi IMAGE-ID
